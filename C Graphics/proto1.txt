#include <graphics.h>
#include <stdlib.h>
#include <time.h>
#include <dos.h>

void drawScene() {
    // Sky background
    setfillstyle(SOLID_FILL, LIGHTBLUE);
    bar(0, 0, getmaxx(), getmaxy());

    // Clouds
    setfillstyle(SOLID_FILL, WHITE);
    fillellipse(150, 100, 70, 40);
    fillellipse(220, 100, 80, 50);
    fillellipse(300, 100, 70, 40);

    fillellipse(500, 100, 80, 50);
    fillellipse(600, 100, 70, 40);
    fillellipse(700, 100, 80, 50);

    // Ground
    setfillstyle(SOLID_FILL, BROWN);
    bar(0, 400, getmaxx(), getmaxy());

    // Tree
    setcolor(BROWN);
    rectangle(200, 300, 230, 500);
    setfillstyle(SOLID_FILL, BROWN);
    floodfill(215, 350, BROWN);

    setcolor(GREEN);
    ellipse(215, 250, 0, 360, 80, 100);
    setfillstyle(SOLID_FILL, GREEN);
    floodfill(215, 250, GREEN);

    // House
    setfillstyle(SOLID_FILL, YELLOW);
    bar(500, 300, 700, 500);
    setfillstyle(SOLID_FILL, RED);
    bar(450, 200, 750, 300);
    line(450, 300, 600, 150);
    line(600, 150, 750, 300);

    // Door
    setfillstyle(SOLID_FILL, BROWN);
    bar(570, 400, 630, 500);

    // Window
    setfillstyle(SOLID_FILL, WHITE);
    bar(650, 350, 700, 400);
    setcolor(BLACK);
    line(650, 375, 700, 375);
    line(675, 350, 675, 400);
}

void drawRain() {
    for (int i = 0; i < 100; i++) {
        int x = rand() % getmaxx();
        int y = rand() % getmaxy();
        setcolor(WHITE);
        line(x, y, x, y + 10);
    }
}

void fillWater(int waterLevel) {
    setfillstyle(SOLID_FILL, BLUE);
    bar(0, 500 - waterLevel, getmaxx(), getmaxy());
}

int main() {
    int gd = DETECT, gm;
    initgraph(&gd, &gm, "");
    srand(time(NULL));

    int waterLevel = 0;

    while (!kbhit()) {
        cleardevice();
        drawScene();
        drawRain();
        fillWater(waterLevel);

        // Increase water level gradually
        if (waterLevel < 100)
            waterLevel++;

        delay(50);
    }

    getch();
    closegraph();
    return 0;
}
